//Use this script to load external JS scripts for IE8 only.
//This is a sniffer script that detects an IE8 browser and loads a specific IE8 combined script
//If it is not an IE8 browser, it loads another script.
//Not sure if this method allows the caching of the JS file - I highly doubt it.

<script type="text/javascript">
    //detect if browser is IE8 and display console message
        if (navigator.userAgent.match(/MSIE 8/) !== null) {
              var script = document.createElement('script');
              script.type = 'text/javascript';
              script.src = './?a=344387:dist/js/footer-main-min-ie8.js';
              document.body.appendChild(script);
        } else { //load combined/min scripts for modern browsers
              var script = document.createElement('script');
              script.type = 'text/javascript';
              script.src = './?a=344387:dist/js/footer-main-min.js';
              document.body.appendChild(script);
        }
</script>
